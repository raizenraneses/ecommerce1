10   import java.util.Date;
11   import java.util.Locale;
12   import java.util.Random;
13   
14   public class receiptFile extends AppCompatActivity {
15   
16       @Override
17       protected void onCreate(Bundle savedInstanceState) {
18           super.onCreate(savedInstanceState);
19           setContentView(R.layout.receipt_file);
20   
21           // Initialize all views first
22           TextView orderNumberTextView = findViewById(R.id.orderNumberTextView);
23           TextView orderDateTextView = findViewById(R.id.orderDateTextView);
24           TextView customerNameTextView = findViewById(R.id.customerNameTextView);
25           TextView customerPhoneTextView = findViewById(R.id.customerPhoneTextView);
26           TextView customerAddressTextView = findViewById(R.id.customerAddressTextView);
27           TextView productNameTextView = findViewById(R.id.productNameTextView);
28           TextView productDescriptionTextView = findViewById(R.id.productDescriptionTextView);
29           TextView productPriceTextView = findViewById(R.id.productPriceTextView);
30           TextView productQuantityTextView = findViewById(R.id.productQuantityTextView);
31           TextView subtotalTextView = findViewById(R.id.subtotalTextView);
32           TextView deliveryFeeTextView = findViewById(R.id.deliveryFeeTextView);
33           TextView totalPriceTextView = findViewById(R.id.totalPriceTextView);
34           TextView deliveryTimeTextView = findViewById(R.id.deliveryTimeTextView);
35           Button trackOrderButton = findViewById(R.id.trackOrderButton);
36           Button backToHomeButton = findViewById(R.id.backToHomeButton);
37   
38           // Get data from intent with null checks
39           Intent intent = getIntent();
40           if (intent == null) {
41               Toast.makeText(this, "Error: No order data received", Toast.LENGTH_SHORT).show();
42               finish();
43               return;
44           }
45   
46           try {
47               String productName = intent.getStringExtra("productName");
48               String productDescription = intent.getStringExtra("productDescription");
49               int quantity = intent.getIntExtra("quantity", 1);
50               int productPrice = intent.getIntExtra("productPrice", 300);
51               int deliveryFee = intent.getIntExtra("deliveryFee", 59);
52               String customerName = intent.getStringExtra("customerName");
53               String customerPhone = intent.getStringExtra("customerPhone");
54               String customerAddress = intent.getStringExtra("customerAddress");
55   
56               // Validate received data
57               if (productName == null || productDescription == null || customerName == null ||
58                       customerPhone == null || customerAddress == null) {
59                   throw new Exception("Invalid order data");
60               }
61   
62               // Calculate totals
63               int subtotal = productPrice * quantity;
64               int total = subtotal + deliveryFee;
65   
66               // Generate random order number
67               String orderNumber = "ORD-" + (100000 + new Random().nextInt(900000));
68   
69               // Get current date and time
70               SimpleDateFormat dateFormat = new SimpleDateFormat("MMM dd, yyyy", Locale.getDefault());
71               SimpleDateFormat timeFormat = new SimpleDateFormat("hh:mm a", Locale.getDefault());
72               String currentDate = dateFormat.format(new Date());
73               String currentTime = timeFormat.format(new Date());
74   
75               // Set values to views
76               orderNumberTextView.setText(orderNumber);
77               orderDateTextView.setText(String.format("%s, %s", currentDate, currentTime));
78               customerNameTextView.setText(customerName);
79               customerPhoneTextView.setText(customerPhone);
80               customerAddressTextView.setText(customerAddress);
81               productNameTextView.setText(productName);
82               productDescriptionTextView.setText(productDescription);
83               productPriceTextView.setText(String.format("P%d", productPrice));
84               productQuantityTextView.setText(String.format("x%d", quantity));
85               subtotalTextView.setText(String.format("P%d", subtotal));
86               deliveryFeeTextView.setText(String.format("P%d", deliveryFee));
87               totalPriceTextView.setText(String.format("P%d", total));
88               deliveryTimeTextView.setText("STANDARD (40-45 mins)");
89   
90               // Set button click listeners
91               trackOrderButton.setOnClickListener(v -> {
92                   Toast.makeText(this, "Tracking your order...", Toast.LENGTH_SHORT).show();
93                   // Implement actual tracking functionality here
94               });
95   
96               backToHomeButton.setOnClickListener(v -> navigateToHome());
97   
98           } catch (Exception e) {
99               Toast.makeText(this, "Error processing order: " + e.getMessage(), Toast.LENGTH_SHORT).show();
100              finish();
101          }
102      }
103  
104      private void navigateToHome() {
105          Intent homeIntent = new Intent(this, brgHomepage.class);
106          homeIntent.addFlags(Intent.FLAG_ACTIVITY_CLEAR_TOP | Intent.FLAG_ACTIVITY_NEW_TASK);
107          startActivity(homeIntent);
108          finish();
109      }
