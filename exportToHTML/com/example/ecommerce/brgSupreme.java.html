1    package com.example.ecommerce;
2    
3    import android.content.Intent;
4    import android.os.Bundle;
5    import android.widget.Button;
6    import android.widget.TextView;
7    import android.widget.Toast;
8    import androidx.appcompat.app.AppCompatActivity;
9    
10   public class brgSupreme extends AppCompatActivity {
11   
12       // Constants
13       private static final int MIN_QUANTITY = 1;
14       private static final int MAX_QUANTITY = 20;
15       private static final int PRODUCT_PRICE = 500;
16       private static final int DELIVERY_FEE = 59;
17   
18       // Views
19       private TextView quantityTextView;
20       private TextView deliveryFeeTextView;
21       private TextView totalPriceTextView;
22       private Button decreaseButton;
23       private Button increaseButton;
24       private TextView cancelButton;
25       private TextView placeOrderButton;
26   
27       // Variables
28       private int quantity = 1;
29   
30       @Override
31       protected void onCreate(Bundle savedInstanceState) {
32           super.onCreate(savedInstanceState);
33           setContentView(R.layout.activity_brg_supreme);
34   
35           // Initialize views
36           quantityTextView = findViewById(R.id.quantityTextView);
37           deliveryFeeTextView = findViewById(R.id.deliveryFeeTextView);
38           totalPriceTextView = findViewById(R.id.totalPriceTextView);
39           decreaseButton = findViewById(R.id.decreaseButton);
40           increaseButton = findViewById(R.id.increaseButton);
41           cancelButton = findViewById(R.id.cancelButton);
42           placeOrderButton = findViewById(R.id.placeOrderButton);
43   
44           // Set initial values
45           updateQuantityDisplay();
46           updatePriceDisplay();
47   
48           // Set click listeners
49           decreaseButton.setOnClickListener(v -> decreaseQuantity());
50   
51           increaseButton.setOnClickListener(v -> increaseQuantity());
52   
53           cancelButton.setOnClickListener(v -> {
54               Toast.makeText(brgSupreme.this, "Order cancelled", Toast.LENGTH_SHORT).show();
55               finish();
56           });
57   
58           placeOrderButton.setOnClickListener(v -> placeOrder());
59       }
60   
61       private void decreaseQuantity() {
62           if (quantity > MIN_QUANTITY) {
63               quantity--;
64               updateQuantityDisplay();
65               updatePriceDisplay();
66           } else {
67               Toast.makeText(this, "Minimum quantity reached", Toast.LENGTH_SHORT).show();
68           }
69       }
70   
71       private void increaseQuantity() {
72           if (quantity < MAX_QUANTITY) {
73               quantity++;
74               updateQuantityDisplay();
75               updatePriceDisplay();
76           } else {
77               Toast.makeText(this, "Maximum quantity reached", Toast.LENGTH_SHORT).show();
78           }
79       }
80   
81       private void updateQuantityDisplay() {
82           quantityTextView.setText(String.valueOf(quantity));
83       }
84   
85       private void updatePriceDisplay() {
86           int total = calculateTotalPrice();
87           deliveryFeeTextView.setText(String.format("P%d", DELIVERY_FEE));
88           totalPriceTextView.setText(String.format("P%d", total));
89       }
90   
91       private void placeOrder() {
92           Intent receiptIntent = new Intent(brgSupreme.this, receiptFileOne.class);
93   
94           // Pass all necessary data to receipt
95           receiptIntent.putExtra("productName", "Supreme");
96           receiptIntent.putExtra("productDescription", "Smoky BBQ sauce, Crispy Bacon");
97           receiptIntent.putExtra("quantity", quantity);
98           receiptIntent.putExtra("productPrice", PRODUCT_PRICE);
99           receiptIntent.putExtra("deliveryFee", DELIVERY_FEE);
100          receiptIntent.putExtra("customerName", "Nina Hana Chang");
101          receiptIntent.putExtra("customerPhone", "09334678901");
102          receiptIntent.putExtra("customerAddress", "Upper 3, Sta Monica, Penacincio, Calumpang, Antipolo City, Bayani");
103  
104          startActivity(receiptIntent);
105          Toast.makeText(this,
106                  String.format("Order placed for %d items! Total: P%d", quantity, calculateTotalPrice()),
107                  Toast.LENGTH_SHORT).show();
108      }
109  
110      private int calculateTotalPrice() {
111          return (PRODUCT_PRICE * quantity) + DELIVERY_FEE;
112      }
113  }
