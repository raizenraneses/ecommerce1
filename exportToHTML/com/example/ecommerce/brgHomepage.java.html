1    package com.example.ecommerce;
2    
3    import android.content.Intent;
4    import android.os.Bundle;
5    import android.widget.Button;
6    import android.widget.Toast;
7    
8    import androidx.appcompat.app.AppCompatActivity;
9    
10   import com.google.firebase.auth.FirebaseAuth;
11   
12   public class brgHomepage extends AppCompatActivity {
13   
14       private Button logoutButton, supreme, baconCheese, classic;
15       private FirebaseAuth mAuth;
16   
17       @Override
18       protected void onCreate(Bundle savedInstanceState) {
19           super.onCreate(savedInstanceState);
20           setContentView(R.layout.activity_brg_homepage);
21   
22           // Initialize FirebaseAuth
23           mAuth = FirebaseAuth.getInstance();
24   
25           // Check if user is logged in
26           if (mAuth.getCurrentUser() == null) {
27               // User not logged in, redirect to login form
28               startActivity(new Intent(brgHomepage.this, loginForm.class));
29               finish();
30               return;
31           }
32   
33           // Initialize buttons
34           logoutButton = findViewById(R.id.logoutButton);
35           supreme = findViewById(R.id.supreme);
36           baconCheese = findViewById(R.id.baconCheese);
37           classic = findViewById(R.id.classic);
38   
39           // Burger button listeners
40           supreme.setOnClickListener(v -> startActivity(new Intent(this, brgSupreme.class)));
41           baconCheese.setOnClickListener(v -> startActivity(new Intent(this, brgBaconCheese.class)));
42           classic.setOnClickListener(v -> startActivity(new Intent(this, brgClassic.class)));
43   
44           // Logout button listener
45           logoutButton.setOnClickListener(v -> {
46               mAuth.signOut();
47               Toast.makeText(this, "Logged out.", Toast.LENGTH_SHORT).show();
48               startActivity(new Intent(this, loginForm.class));
49               finish();
50           });
51       }
52   }
