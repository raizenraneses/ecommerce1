1    package com.example.ecommerce;
2    
3    import android.content.Intent;
4    import android.os.Bundle;
5    import android.text.TextUtils;
6    import android.util.Patterns;
7    import android.widget.Button;
8    import android.widget.TextView;
9    import android.widget.Toast;
10   
11   import androidx.appcompat.app.AppCompatActivity;
12   
13   import com.google.android.material.textfield.TextInputEditText;
14   import com.google.firebase.auth.FirebaseAuth;
15   
16   public class loginForm extends AppCompatActivity {
17   
18       private TextInputEditText etEmail, etPassword;
19       private Button btnLogin;
20       private TextView tvCreateAccount;
21   
22       private FirebaseAuth mAuth;
23   
24       @Override
25       protected void onCreate(Bundle savedInstanceState) {
26           super.onCreate(savedInstanceState);
27           setContentView(R.layout.activity_login_form);
28   
29           etEmail = findViewById(R.id.etEmail);
30           etPassword = findViewById(R.id.etPassword);
31           btnLogin = findViewById(R.id.btnLogin);
32           tvCreateAccount = findViewById(R.id.tvCreateAccount);
33   
34           mAuth = FirebaseAuth.getInstance();
35   
36           btnLogin.setOnClickListener(v -> attemptLogin());
37           tvCreateAccount.setOnClickListener(v -> {
38               Intent intent = new Intent(this, createAccount.class);
39               startActivity(intent);
40               finish();
41           });
42       }
43   
44       private void attemptLogin() {
45           String email = etEmail.getText().toString().trim();
46           String password = etPassword.getText().toString().trim();
47   
48           if (TextUtils.isEmpty(email) || !Patterns.EMAIL_ADDRESS.matcher(email).matches()) {
49               etEmail.setError("Valid email is required");
50               etEmail.requestFocus();
51               return;
52           }
53   
54           if (TextUtils.isEmpty(password)) {
55               etPassword.setError("Password is required");
56               etPassword.requestFocus();
57               return;
58           }
59   
60           mAuth.signInWithEmailAndPassword(email, password)
61                   .addOnCompleteListener(task -> {
62                       if (task.isSuccessful()) {
63                           Toast.makeText(loginForm.this, "Login successful", Toast.LENGTH_SHORT).show();
64                           startActivity(new Intent(loginForm.this, brgHomepage.class));
65                           finish();
66                       } else {
67                           Toast.makeText(loginForm.this, "Login failed: " + task.getException().getMessage(), Toast.LENGTH_LONG).show();
68                       }
69                   });
70   
71   
72       }
73   }
